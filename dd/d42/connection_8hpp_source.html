49 inline void write_packet(Output& output, std::error_code& ec,
50  const Connect_header<String, Will_message, Password>& header)
51 {
52  if (!header.will && (header.will_qos != QoS::at_most_once || header.will_retain)) {
53  ec = Error::bad_will;
54  return;
55  } else if (!header.username && header.password) {
56  ec = Error::bad_username_password;
57  return;
58  } else if (!header.clean_session && !header.client_identifier.size()) {
59  ec = Error::empty_client_identifier;
60  return;
61  }
62 
63  // fixed header & protocol name & level & connect flags
64  typename std::underlying_type<Variable_integer>::type remaining =
65  12 + (header.will ? 4 : 0) + (header.username ? 2 : 0) + (header.password ? 2 : 0);
66  const auto protocol_level = Byte{ 0x04 };
67  const auto connect_flags = static_cast<Byte>(
68  (header.username ? 0x80 : 0x00) | (header.password ? 0x40 : 0x00) | (header.will_retain << 5) |
69  (static_cast<int>(header.will_qos) << 3) | (header.will ? 0x04 : 0x00) | (header.clean_session << 1));
70 
71  if (!protected_add(remaining, header.client_identifier.size()) ||
72  (header.will && !protected_add(remaining, header.will->first.size())) ||
73  (header.will && !protected_add(remaining, header.will->second.size())) ||
74  (header.username && !protected_add(remaining, header.username->size())) ||
75  (header.password && !protected_add(remaining, header.password->size()))) {
76  ec = Error::payload_too_large;
77  return;
78  }
79 
80  write_elements(output, ec, Byte{ static_cast<int>(Control_packet_type::connect) << 4 },
81  static_cast<Variable_integer>(remaining), std::uint16_t{ 4 }, Byte{ 'M' }, Byte{ 'Q' },
82  Byte{ 'T' }, Byte{ 'T' }, protocol_level, connect_flags, header.keep_alive);
83 
84  if (ec) {
85  return;
86  }
87 
88  // payload
89  if (write_blob<true>(output, ec, header.client_identifier), ec) {
90  return;
91  }
92 
93  if (header.will) {
94  if (write_blob<true>(output, ec, header.will->first), ec) {
95  return;
96  }
97 
98  if (write_blob<true>(output, ec, header.will->second), ec) {
99  return;
100  }
101  }

116 template<typename Input>
117 inline bool read_packet(Input& input, std::error_code& ec, Read_context& context, Connack_header& header)
118 {
119  if (context.sequence == 0) {
120  Byte type;
121  if (!read_element(input, ec, context, type) || ec) {
122  return false;
123  } else if (type != static_cast<Byte>(static_cast<int>(Control_packet_type::connack) << 4)) {
124  ec = Error::bad_connack_flags;
125  return false;
126  }
127  }
128 
129  if (context.sequence == 1) {
130  Variable_integer remaining;
131  if (!read_element(input, ec, context, remaining) || ec) {
132  return false;
133  } else if (remaining != static_cast<Variable_integer>(2)) {
134  ec = Error::bad_connack_payload;
135  return false;
136  }
137  }
138 
139  if (context.sequence == 2) {
140  Byte flags;
141  if (!read_element(input, ec, context, flags) || ec) {
142  return false;
143  } else if (flags & 0xfe) {
144  ec = Error::bad_connack_flags;
145  return false;
146  }
147  header.session_present = flags & 0x01;
148  }
149 
150  if (context.sequence == 3) {
151  Byte rc;
152  if (!read_element(input, ec, context, rc) || ec) {
153  return false;
154  } else if (rc > static_cast<Byte>(Connack_return_code::not_authorized)) {
155  ec = Error::bad_connack_return_code;
156  return false;
157  }
158  header.return_code = static_cast<Connack_return_code>(rc);
159  }
160 
161  return true;
162 }
